概述

在数论中，如果一个线性同余方程 ，则我们就说 和 在模 意义下互为乘法逆元，记作 或 。

背景

那么乘法逆元有什么用呢？在关于除法的取模运算中将会应用到。

根据同余关系，可以得到在加法和乘法场景下的两条重要推论，在进行模运算时，进行处处取模操作以避免数值溢出： 而除法却不满足这样的性质；具体的推导和证明请看这篇文章 。

比如，加法运算取模：

(12 + 3) % 10 = 5
可以转换为：
(12 % 10 + 3 % 10) % 10
= (2 + 3) % 10
= 5

然而，除法运算取模就不行了：

(12 / 3) % 10 = 4
如果转换为：
((12 % 10) / (3 % 10)) % 10
= (2 / 3) % 10
???
(2 / 3) 无法得到整数结果，这就不符合我们的要求了

因此，我们将引入乘法逆元，将除法操作转换为乘法，以解决这个问题：

(12 / 3) % 10
引入逆元可以转换为:
(12 * inverse(3)) % 10
则有:
(2 * (inverse(3) % 10)) % 10
我们只要计算 inverse(3) 就可以完成计算了

因此，以下将介绍如何求逆元。

/**
 * exgcd 求逆元，当且仅当 gcd(a,m) = 1 时才有解。
 * @param a
 * @param m
 * @return
 */
public static int inverse(int a, int m) {
    int[] res = exgcd(a, m);
    int gcd = res[0];
    int x = res[1];
    if (gcd != 1) {
        // No solution
        return -1;
    } else {
        return (x % m + m) % m;
    }
}

public static int[] exgcd(int a, int b) {
    if (b == 0) {
        return new int[]{a, 1, 0};
    }
    int[] res = exgcd(b, a % b);
    int g = res[0];
    int x = res[1];
    int y = res[2];
    return new int[]{g, y, x - y * (a / b)};
}

public static int qpow(int a, int p, int m) {
    int ans = 1;
    while (p > 0) {
        if ((p & 1) == 1) {
            ans = (int)(((long)ans * a) % m);
        }
        a = (int)(((long)a * a) % m);
        p >>= 1;
    }
    return ans;
}

递归实现

使用递归计算的同时记忆化保存每个逆元，可以以 的复杂度完成 所有数的计算。

public class LinearInverse {
    private final int a;
    private final int m;
    private final int[] inverseArr;
    public LinearInverse(int a, int m) {
        this.a = a;
        this.m = m;
        this.inverseArr = new int[a + 1];
        Arrays.fill(this.inverseArr, -1);
    }

    private static int mod(int a, int m) {
        return (a % m + m) % m;
    }

    private static int longMod(long a, int m) {
        return (int)(a % m);
    }

    /**
     * 线性求逆元，计算 [1,2,...,a] 中所有数的逆元；
     * <p> 递归实现，用 inverseArr[] 进行记忆化，因此总复杂度为 O(n)
     * @param a
     * @param m
     * @return
     */
    public int[] inverseAll() {
        for (int i = 1; i <= a; i++) {
            if (inverseArr[i] != -1) continue;
            inverse(i, m);
        }
        return this.inverseArr;
    }

    /**
     * 递归计算单个逆元
     * @param a
     * @param m
     * @return
     */
    private int inverse(int a, int m) {
        if (a <= 1) {
            inverseArr[a] = a;
            return a;
        }
        inverseArr[a] = mod(longMod((long)(m - m / a) * (inverse(mod(m, a), m)), m), m);
        return inverseArr[a];
    }

    public static void main(String[] args) {
        int a = 100;
        int m = (int)(1e9 + 7);
        LinearInverse linearInverse = new LinearInverse(a, m);
        linearInverse.inverseAll();
        System.out.println(Arrays.toString(linearInverse.inverseArr));
    }
}

/**

 * <p> 迭代实现，总复杂度为 O(n)
 * @param a

public static int[] linearInverse(int a, int m) {
    int[] inverseArr = new int[a + 1];
    inverseArr[1] = 1;
    for (int i = 2; i <= a; i++) {
        inverseArr[i] = mod(longMod((long)(m - (m / i)) * inverseArr[m % i], m), m);
    }
    return inverseArr;
}

/**
 * 线性求逆元，计算数组 a[i] 中所有数的逆元；
 * <p> 总复杂度为 O(n)
 * @param a
 * @param m
 * @return
 */
public static int[] linearInverseArray(int[] a, int m) {
    int n = a.length;
    int[] res = new int[n];
    int[] prefix = new int[n + 1];
    prefix[0] = 1;
    for (int i = 1; i <= n; i++) {
        prefix[i] = longMod((long)prefix[i - 1] * a[i - 1], m);
    }
    int[] invInfo = exgcd(prefix[n], m);
    int invAll = (invInfo[1] % m + m) % m; // x
    int current = invAll;
    for (int i = n - 1; i >= 0; i--) {
        res[i] = longMod((long)prefix[i] * current, m);
        current = longMod((long)current * a[i], m);
    }
    return res;
}

private static int longMod(long a, int m) {
    return (int)(a % m);
}

public static int[] exgcd(int a, int b) {
    if (b == 0) {
        return new int[]{a, 1, 0};
    }
    int[] res = exgcd(b, a % b);
    int g = res[0];
    int x = res[1];
    int y = res[2];
    return new int[]{g, y, x - y * (a / b)};
}
